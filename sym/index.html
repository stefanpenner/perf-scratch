function intern(str) {
  var obj = {};
  obj[str] = 1;
  for (var key in obj) {
    if (key === str) {
      return key;
    }
  }
  return str;
}

function symbol(debug) {
  return intern(debug + ' [id=' + GUID_KEY + Math.floor(Math.random() * new Date()) + ']');
}
